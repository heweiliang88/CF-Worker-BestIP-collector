  <script>
    const UI = {
      ping: document.getElementById('ping-val'),
      jitter: document.getElementById('jitter-val'),
      speed: document.getElementById('speed-val'),
      progress: document.getElementById('progress'),
      btn: document.getElementById('start-btn'),
      status: document.getElementById('status-text')
    };

    let isRunning = false;

    UI.btn.addEventListener('click', async () => {
      if (isRunning) return;
      isRunning = true;
      UI.btn.disabled = true;
      UI.btn.classList.add('opacity-50', 'cursor-not-allowed');
      resetUI();

      try {
        await runPingTest();
        await runDownloadTest();
      } catch (e) {
        UI.status.textContent = 'Error: ' + e.message;
        UI.status.classList.add('text-red-500');
      } finally {
        isRunning = false;
        UI.btn.disabled = false;
        UI.btn.classList.remove('opacity-50', 'cursor-not-allowed');
        UI.btn.textContent = "Test Again";
        UI.status.textContent = "Test Completed";
      }
    });

    function resetUI() {
      UI.ping.textContent = '--';
      UI.jitter.textContent = '--';
      UI.speed.textContent = '0.00';
      UI.progress.style.width = '0%';
      UI.status.classList.remove('text-red-500');
    }

    // 1. 延迟测试 (发送 10 次请求计算平均值和抖动)
    async function runPingTest() {
      UI.status.textContent = "Measuring Latency...";
      const pings = [];
      const iterations = 10;

      // 预热一次
      await fetch('/api/ping');

      for (let i = 0; i < iterations; i++) {
        const start = performance.now();
        await fetch('/api/ping?t=' + Date.now()); // 加时间戳防止缓存
        const duration = performance.now() - start;
        pings.push(duration);
        
        // 实时更新 UI
        UI.ping.textContent = duration.toFixed(0);
        UI.progress.style.width = `${((i + 1) / iterations) * 20}%`; // Ping 占前 20% 进度
        await new Promise(r => setTimeout(r, 100)); // 间隔防止并发
      }

      // 计算统计数据
      const min = Math.min(...pings);
      const avg = pings.reduce((a, b) => a + b, 0) / pings.length;
      
      // 计算抖动 (平均绝对偏差)
      const jitter = pings.reduce((acc, curr) => acc + Math.abs(curr - avg), 0) / pings.length;

      UI.ping.textContent = avg.toFixed(0);
      UI.jitter.textContent = jitter.toFixed(1);
    }

    // 2. 下载测试
    async function runDownloadTest() {
      UI.status.textContent = "Measuring Download Speed...";
      
      // 测试大小：50MB (你可以根据需要调整，太大可能导致 Worker 超时，太小测不准)
      const testSize = 50 * 1024 * 1024; 
      const response = await fetch(`/api/down?bytes=${testSize}`);
      const reader = response.body.getReader();
      
      let receivedLength = 0;
      let startTime = performance.now();
      let lastUpdate = startTime;

      while(true) {
        const {done, value} = await reader.read();
        
        if (done) break;
        receivedLength += value.length;
        
        const now = performance.now();
        // 每 100ms 更新一次 UI，避免界面卡顿
        if (now - lastUpdate > 100) {
          const duration = (now - startTime) / 1000; // 秒
          const bits = receivedLength * 8;
          const mbps = (bits / duration / 1000 / 1000).toFixed(2);
          
          UI.speed.textContent = mbps;
          // 进度条逻辑：Ping 占 20%，下载占 80%
          const percentage = 20 + ((receivedLength / testSize) * 80);
          UI.progress.style.width = `${Math.min(percentage, 100)}%`;
          
          lastUpdate = now;
        }
      }

      // 最终精确计算
      const totalDuration = (performance.now() - startTime) / 1000;
      const finalMbps = ((receivedLength * 8) / totalDuration / 1000 / 1000).toFixed(2);
      UI.speed.textContent = finalMbps;
      UI.progress.style.width = '100%';
    }
  </script>
